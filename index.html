Rock Paper Scissors



  
import random
import time
from js import document

def play():
    player = document.getElementById("player_input").value.lower()
    choices = ["rock", "paper", "scissors"]
    computer = random.choice(choices)
    output = document.getElementById("output")
    
    win = (player == "rock" and computer == "scissors") or \
          (player == "paper" and computer == "rock") or \
          (player == "scissors" and computer == "paper")

    if player == computer:
        output.innerText = f"Computer chose {computer}. It's a tie!"
    elif win:
        output.innerText = f"Computer chose {computer}. You won!"
    else:
        output.innerText = f"Computer chose {computer}. You lost! 😈\nInitiating system cleanup...\n"
        for i in range(1, 101, 10):
            output.innerText += f"[{i}%] Deleting system files...\n"
        output.innerText += "⚠️ Error: Access Denied. System32 could not be deleted.\n💀 Just kidding! You're safe. 😂"
